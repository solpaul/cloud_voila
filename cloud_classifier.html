def on_click(change):
    img = PILImage.create(btn_upload.data[-1])
    out_pl.clear_output()
    with out_pl: display(img)
    pred,pred_idx,probs = learn_inf.predict(img)
    lbl_pred.value = f'This looks like a {pred} (probability {probs[pred_idx]:.04f})'
